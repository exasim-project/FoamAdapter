76 void assemble(NeoN::scalar t, NeoN::scalar dt)
77 {
78 auto vol = psi_.mesh().cellVolumes().view();
79 auto expSource = expr_.explicitOperation(psi_.mesh().nCells());
80 expr_.explicitOperation(expSource, t, dt);
81 auto expSourceView = expSource.view();
82 fill(ls_.rhs(), NeoN::zero<ValueType>());
83 fill(ls_.matrix().values(), NeoN::zero<ValueType>());
84 expr_.implicitOperation(ls_);
85 // TODO rename implicitOperation -> assembleLinearSystem
86 expr_.implicitOperation(ls_, t, dt);
87 auto rhs = ls_.rhs().view();
88 // we subtract the explicit source term from the rhs
89 NeoN::parallelFor(
90 exec(),
91 {0, rhs.size()},
92 KOKKOS_LAMBDA(const NeoN::localIdx i) { rhs[i] -= expSourceView[i] * vol[i]; }
93 );

155 void setReference(const IndexType refCell, ValueType refValue)
156 {
157 // TODO currently assumes that matrix is already assembled
158 const auto diagOffset = sparsityPattern_.diagOffset().view();
159 const auto rowOffs = ls_.matrix().rowOffs().view();
160 auto rhs = ls_.rhs().view();
161 auto values = ls_.matrix().values().view();
162 NeoN::parallelFor(
163 ls_.exec(),
164 {refCell, refCell + 1},
165 KOKKOS_LAMBDA(const std::size_t refCelli) {
166 auto diagIdx = rowOffs[refCelli] + diagOffset[refCelli];
167 auto diagValue = values[diagIdx];
168 rhs[refCelli] += diagValue * refValue;
169 values[diagIdx] += diagValue;
170 }
171 );
172 }

184template<typename ValueType, typename IndexType = NeoN::localIdx>
185NeoN::Vector<ValueType> diag(
186 const la::LinearSystem<ValueType, IndexType>& ls,
187 const NeoN::la::SparsityPattern& sparsityPattern
188)
189{
190 NeoN::Vector<ValueType> diagonal(ls.exec(), sparsityPattern.diagOffset().size(), 0.0);
191 auto diagView = diagonal.view();
192
193 const auto diagOffset = sparsityPattern.diagOffset().view();
194 const auto [matrix, b] = ls.view();
195 NeoN::parallelFor(
196 ls.exec(),
197 {0, diagOffset.size()},
198 KOKKOS_LAMBDA(const std::size_t celli) {
199 auto diagOffsetCelli = diagOffset[celli];
200 diagView[celli] = matrix.values[matrix.rowOffs[celli] + diagOffsetCelli];
201 }
202 );
203 return diagonal;
204}

207template<typename ValueType, typename IndexType = NeoN::localIdx>
208NeoN::finiteVolume::cellCentred::VolumeField<ValueType> applyOperator(
209 const la::LinearSystem<ValueType, IndexType>& ls,
210 const NeoN::finiteVolume::cellCentred::VolumeField<ValueType>& psi
211)
212{
213 NeoN::finiteVolume::cellCentred::VolumeField<ValueType> resultVector(
214 psi.exec(),
215 "ls_" + psi.name,
216 psi.mesh(),
217 psi.internalVector(),
218 psi.boundaryData(),
219 psi.boundaryConditions()
220 );
221
222 auto [result, x] = views(resultVector.internalVector(), psi.internalVector());
223 const auto [matrix, b] = ls.view();
224
225 NeoN::parallelFor(
226 resultVector.exec(),
227 {0, result.size()},
228 KOKKOS_LAMBDA(const std::size_t rowi) {
229 IndexType rowStart = matrix.rowOffs[rowi];
230 IndexType rowEnd = matrix.rowOffs[rowi + 1];
231 ValueType sum = 0.0;
232 for (IndexType coli = rowStart; coli < rowEnd; coli++)
233 {
234 sum += matrix.values[coli] * x[matrix.colIdxs[coli]];
235 }
236 result[rowi] = sum - b[rowi];
237 }
238 );
239
240 return resultVector;
241}
