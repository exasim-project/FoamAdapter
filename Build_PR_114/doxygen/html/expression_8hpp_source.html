85 struct SetReference {

94 void operator()(NeoN::la::LinearSystem<FunctorValueType, NeoN::localIdx>& ls){
95
96 const auto diagOffset = ls.sparsityPattern().diagOffset().view();
97 const auto rowOffs = ls.matrix().rowOffs().view();
98 auto rhs = ls.rhs().view();
99 auto values = ls.matrix().values().view();
100
101 NeoN::parallelFor(
102 ls.exec(),
103 {pRefCell_, pRefCell_ + 1},
104 KOKKOS_LAMBDA(const std::size_t refCelli) {
105 auto diagIdx = rowOffs[refCelli] + diagOffset[refCelli];
106 auto diagValue = values[diagIdx];
107 rhs[refCelli] += diagValue * pRefValue_;
108 values[diagIdx] += diagValue;
109 }
110 );
111
112 }
113 };

155template<typename ValueType, typename IndexType = NeoN::localIdx>
156NeoN::Vector<ValueType> diag(
157 const la::LinearSystem<ValueType, IndexType>& ls,
158 const NeoN::la::SparsityPattern& sparsityPattern
159)
160{
161 NeoN::Vector<ValueType> diagonal(ls.exec(), sparsityPattern.diagOffset().size(), 0.0);
162 auto diagView = diagonal.view();
163
164 const auto diagOffset = sparsityPattern.diagOffset().view();
165 const auto [matrix, b] = ls.view();
166 NeoN::parallelFor(
167 ls.exec(),
168 {0, diagOffset.size()},
169 KOKKOS_LAMBDA(const std::size_t celli) {
170 auto diagOffsetCelli = diagOffset[celli];
171 diagView[celli] = matrix.values[matrix.rowOffs[celli] + diagOffsetCelli];
172 }
173 );
174 return diagonal;
175}

178template<typename ValueType, typename IndexType = NeoN::localIdx>
179NeoN::finiteVolume::cellCentred::VolumeField<ValueType> applyOperator(
180 const la::LinearSystem<ValueType, IndexType>& ls,
181 const NeoN::finiteVolume::cellCentred::VolumeField<ValueType>& psi
182)
183{
184 NeoN::finiteVolume::cellCentred::VolumeField<ValueType> res(
185 psi.exec(),
186 "ls_" + psi.name,
187 psi.mesh(),
188 psi.internalVector(),
189 psi.boundaryData(),
190 psi.boundaryConditions()
191 );
192
193 NeoN::la::computeResidual(
194 ls.matrix(), ls.rhs(), psi.internalVector(), res.internalVector());
195 return res;
196}
