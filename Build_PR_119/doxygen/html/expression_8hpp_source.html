78 template<typename FunctorValueType>
79 struct SetReference : public NeoN::dsl::PostAssemblyBase<ValueType>
80 {
81
82 NeoN::localIdx pRefCell_;
83 NeoN::scalar pRefValue_;
84
85 SetReference(NeoN::localIdx pRefCell, NeoN::scalar pRefValue)
86 : pRefCell_(pRefCell)
87 , pRefValue_(pRefValue)
88 {}
89
90 virtual void operator()(
91 const NeoN::la::SparsityPattern& sp,
92 NeoN::la::LinearSystem<FunctorValueType, NeoN::localIdx>& ls
93 )
94 {
95 const auto diagOffset = sp.diagOffset().view();
96 const auto rowOffs = ls.matrix().rowOffs().view();
97 auto rhs = ls.rhs().view();
98 auto values = ls.matrix().values().view();
99
100 NeoN::parallelFor(
101 ls.exec(),
102 {pRefCell_, pRefCell_ + 1},
103 KOKKOS_LAMBDA(const std::size_t refCelli) {
104 auto diagIdx = rowOffs[refCelli] + diagOffset[refCelli];
105 auto diagValue = values[diagIdx];
106 rhs[refCelli] += diagValue * pRefValue_;
107 values[diagIdx] += diagValue;
108 }
109 );
110 }
111 };
112
113 void setReference(NeoN::localIdx pRefCell, NeoN::scalar pRefValue)
114 {
115 needReference_ = true;
116 pRefCell_ = pRefCell;
117 pRefValue_ = pRefValue;
118 }

120 // TODO unify with dsl/solver.hpp
121 NeoN::la::SolverStats solve()
122 {
123 // Only if ValueType is scalar
124 auto functs = std::vector<NeoN::dsl::PostAssemblyBase<ValueType>> {};
125
126 if constexpr (std::is_same_v<ValueType, NeoN::scalar>) {
127 functs = needReference_ ? std::vector<NeoN::dsl::PostAssemblyBase<ValueType>> {}
128 : std::vector<NeoN::dsl::PostAssemblyBase<ValueType>> {SetReference<ValueType>(pRefCell_, pRefValue_)};
129 }
130
131 // FIXME TODO this will create the sparsity pattern and potentially the ls
132 // again even if it has been created already
133 auto solverDict = runTime_.fvSolutionDict.get<NeoN::Dictionary>("solvers");
134 auto fieldSolverDict = solverDict.get<NeoN::Dictionary>(psi_.name);
135 auto stats = NeoN::dsl::detail::iterativeSolveImpl(
136 expr_,
137 sparsityPattern_,
138 ls_,
139 psi_,
140 runTime_.t,
141 runTime_.dt,
142 runTime_.fvSchemesDict,
143 fieldSolverDict,
144 functs
145 );
146
147 std::cout
148 << "[NeoN] Solving for " << psi_.name << ":"
149 << " Initial residual: " << stats.initResNorm
150 << " Final residual: " << stats.finalResNorm
151 << " No Iterations: " << stats.numIter
152 << std::endl;
153 return stats;
154 }

174template<typename ValueType, typename IndexType = NeoN::localIdx>
175NeoN::Vector<ValueType> diag(
176 const la::LinearSystem<ValueType, IndexType>& ls,
177 const NeoN::la::SparsityPattern& sparsityPattern
178)
179{
180 NeoN::Vector<ValueType> diagonal(ls.exec(), sparsityPattern.diagOffset().size(), 0.0);
181 auto diagView = diagonal.view();
182
183 const auto diagOffset = sparsityPattern.diagOffset().view();
184 const auto [matrix, b] = ls.view();
185 NeoN::parallelFor(
186 ls.exec(),
187 {0, diagOffset.size()},
188 KOKKOS_LAMBDA(const std::size_t celli) {
189 auto diagOffsetCelli = diagOffset[celli];
190 diagView[celli] = matrix.values[matrix.rowOffs[celli] + diagOffsetCelli];
191 }
192 );
193 return diagonal;
194}

197template<typename ValueType, typename IndexType = NeoN::localIdx>
198NeoN::finiteVolume::cellCentred::VolumeField<ValueType> applyOperator(
199 const la::LinearSystem<ValueType, IndexType>& ls,
200 const NeoN::finiteVolume::cellCentred::VolumeField<ValueType>& psi
201)
202{
203 NeoN::finiteVolume::cellCentred::VolumeField<ValueType> res(
204 psi.exec(),
205 "ls_" + psi.name,
206 psi.mesh(),
207 psi.internalVector(),
208 psi.boundaryData(),
209 psi.boundaryConditions()
210 );
211 NeoN::la::computeResidual(
212 ls.matrix(), ls.rhs(), psi.internalVector(), res.internalVector());
213 return res;
214}
